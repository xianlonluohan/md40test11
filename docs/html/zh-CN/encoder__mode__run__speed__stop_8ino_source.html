75void loop() {
76 if (g_motor_run) {
77 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
78 g_md40[i].RunSpeed(kMotorSpeed);
79 }
80 } else {
81 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
82 g_md40[i].Stop();
83 }
84 }
85
86 if (millis() - g_trigger_time > 2000) {
87 g_trigger_time = millis();
88 g_motor_run = !g_motor_run;
89 Serial.println(g_motor_run ? "Motors Run" : "Motors Stop");
90 }
91
92 if (millis() - g_last_print_time > 200) {
93 g_last_print_time = millis();
94
95 Serial.print("speeds: ");
96 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
97 Serial.print(g_md40[i].speed());
98 Serial.print(", ");
99 }
100
101 Serial.print("pwm duties: ");
102 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
103 Serial.print(g_md40[i].pwm_duty());
104 Serial.print(", ");
105 }
106
107 Serial.print("positions: ");
108 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
109 Serial.print(g_md40[i].position());
110 Serial.print(", ");
111 }
112
113 Serial.print("pulse counts: ");
114 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
115 Serial.print(g_md40[i].pulse_count());
116 Serial.print(", ");
117 }
118
119 Serial.print("states: ");
120 for (uint8_t i = 0; i < em::Md40::kMotorNum; i++) {
121 Serial.print(static_cast<uint8_t>(g_md40[i].state()));
122 if (i < em::Md40::kMotorNum - 1) {
123 Serial.print(", ");
124 }
125 }
126
127 Serial.println();
128 }
